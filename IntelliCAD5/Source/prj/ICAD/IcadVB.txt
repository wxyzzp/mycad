Type sds_callback_packet
    sds_hdlg  dialog As sds_hdlg;
    sds_htile tile;
    char*     value;
    void*     client_data;
    int       reason;
    long      x, y;

    char *buf;
End Type

Type sds_u_val {
   sds_real rreal;
   sds_real rpoint[3];
   short rint;
   char *rstring;
   long rlname[2];
   long rlong;
   struct sds_binary rbinary;
End Type

struct sds_resbuf {
    struct sds_resbuf *rbnext;
    short restype;
    union sds_u_val resval;
End Type
